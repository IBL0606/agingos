SELECT category,
       COUNT(*) AS n_24h,
       MAX("timestamp") AS max_ts_24h
FROM events
WHERE "timestamp" >= now() - interval '24 hours'
GROUP BY 1
ORDER BY max_ts_24h DESC NULLS LAST;
